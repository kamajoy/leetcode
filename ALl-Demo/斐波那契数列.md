//斐波那契数列-递归 1、1、2、3、5、8、13、21、34
function fib1($n){
    $n = (int)$n;
    if($n <= 2){
        return 1;
    }

    return fib1($n-1) + fib1($n-2);
}

//斐波那契数列-循环 1、1、2、3、5、8、13、21、34
function fib2($n){
    $arr = [];
    for($i=1;$i<=$n;$i++){
        if($i < 3){
            $arr[$i] = 1;
        }else{
            $arr[$i] = $arr[$i-1]+$arr[$i-2];
        }
    }

    return $arr[$n];
}
